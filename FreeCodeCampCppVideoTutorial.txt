message:
        .ascii  "Hello World!\000"

        .ascii  "text: %s\012\000"
main:
        push    {r7, lr}
        sub     sp, sp, #8
        add     r7, sp, #0
        movs    r3, #10
        str     r3, [r7, #4]
        movw    r1, #:lower16:message
        movt    r1, #:upper16:message
        movw    r0, #:lower16:.LC0
        movt    r0, #:upper16:.LC0
        bl      printf
        movs    r3, #0
        mov     r0, r3
        adds    r7, r7, #8
        mov     sp, r7
        pop     {r7, pc}

#include <iostream>
#include <vector>

int main()
{
    std::vector<int> numbers = {1, 2, 3};

    // 1st range based for loop
    //
    // Note that n is a reference in C++, it looks like a pointer but the compiler inserts a dereference under the hood.
    // When you say n *= 2, the compiler translates this to *n *= 2 before it compiles the source code.
    // So a reference in C++ looks like a pointer in C but is not the same,
    // under the hood a pointer line is added and this pointer itself cannot be changed (cannot be reseated).
    for (int& n : numbers)
    {
        n *= 2; // double each element
    }

    // 2nd range based for loop
    for (int n : numbers)
    {
        std::cout << n << " "; // 2 4 6
    }
    std::cout << std::endl;

    return 0;
}

// gpio.h
#ifndef GPIO_H
#define GPIO_H

#include <stdint.h>

typedef struct {
    volatile uint32_t* port;
    uint32_t mask;
} Gpio;

// initialize the "object"
static inline void Gpio_init(Gpio* gpio, volatile uint32_t* port, uint32_t pin) {
    gpio->port = port;
    gpio->mask = (1U << pin);
}

static inline void Gpio_set(Gpio* gpio) {
    *(gpio->port) |= gpio->mask;
}

static inline void Gpio_clear(Gpio* gpio) {
    *(gpio->port) &= ~gpio->mask;
}

static inline void Gpio_toggle(Gpio* gpio) {
    *(gpio->port) ^= gpio->mask;
}

#endif

c++ gpio.hpp

// gpio.hpp
#pragma once
#include <cstdint>

class Gpio {
    volatile uint32_t* port;
    uint32_t mask;

public:
    // explicit init function (instead of constructor)
    void init(volatile uint32_t* port, uint32_t pin) {
        this->port = port;
        this->mask = (1U << pin);
    }

    void set()   { *port |= mask; }
    void clear() { *port &= ~mask; }
    void toggle(){ *port ^= mask; }
};

c++ main.cpp

#include "gpio.hpp"

    Gpio led;
    led.init(PORTA, 5);

    while (1) {
        led.toggle();
        // wait...
    }
}

#include <iostream>
#include <cstdint>

using namespace std;

typedef union
{
    uint8_t     a;
    uint8_t     b;
    uint16_t    c;
    uint32_t    d;
} motor_union;

int main()
{
    string      str;
    float       f1  = 1.0000001;
    float       f2  = 10.000001;
    float       f3  = 100.00001;
    float       f4  = 1000.0001;
    float       f5  = 10000.001;
    float       f6  = 100000.01;
    float       f7  = 1000000.1;
    float       f8  = 0.0000000;
    float       f9  = 0.0000000;
    motor_union motor;

    str = "Gerard Zennipman";
    cout << str << endl;

    printf("Gerard Zennipman\n");

    printf("f1: %15.7f\n", f1);
    printf("f2: %15.7f\n", f2);
    printf("f3: %15.7f\n", f3);
    printf("f4: %15.7f\n", f4);
    printf("f5: %15.7f\n", f5);
    printf("f6: %15.7f\n", f6);
    printf("f7: %15.7f\n", f7);
    printf("f8: %15.7f\n", f8);
    printf("f9: %15.7f\n", f9);

    f8 = 0.0000001;
    f9 = f1 + f8;
    printf("f9: %15.7f\n", f9);

    printf("sizeof(motor): %d\n", sizeof(motor));
    printf("sizeof(motor.a): %d\n", sizeof(motor.a));
    printf("sizeof(motor.b): %d\n", sizeof(motor.b));
    printf("sizeof(motor.c): %d\n", sizeof(motor.c));
    printf("sizeof(motor.d): %d\n", sizeof(motor.d));

    // cout treats uint8_t as a char, cast it to uint32_t to print it as a nr
    motor.a = 100;
    cout << static_cast<uint32_t>(motor.a) << endl;

    // cout treats uint8_t as a char, cast it to uint32_t to print it as a nr
    motor.b = 200;
    cout << static_cast<uint32_t>(motor.b) << endl;

    motor.c = 300;
    cout << motor.c << endl;

    motor.d = 400;
    cout << motor.d << endl;

    // check union behavior
    motor.a = 0x01;
    printf("motor.a: 0x%02X\n", motor.a);
    motor.d = 0x11223344;
    printf("motor.a: 0x%02X\n", motor.a);

    return (0);
}

#include <iostream>

int main()
{
    int a {1};
    int b {2};
    int c {3};
    int d {40};
    int e {5};
    int f {6};
    int g {7};
    int result;

    std::cout << "a: " << a << std::endl;
    std::cout << "b: " << b << std::endl;
    std::cout << "c: " << c << std::endl;
    std::cout << "d: " << d << std::endl;
    std::cout << "e: " << e << std::endl;
    std::cout << "f: " << f << std::endl;
    std::cout << "g: " << g << std::endl;

    result = a + b * c - d / e - f + g;
    std::cout << "a + b * c - d / e - f + g: " << result << std::endl; // 0

    d = 80;
    result = a + b * c - d / e - f + g;
    std::cout << "d: " << d << std::endl;
    std::cout << "a + b * c - d / e - f + g: " << result << std::endl; // -8

    result = d / e * f + g;
    std::cout << "d / e * f + g: " << result << std::endl; // 103

    result = d / (e * f) + g;
    std::cout << "d / (e * f) + g: " << result << std::endl; // 9

    return 0;

#include <iostream>

const size_t STUDENT_COUNT {8};

int main()
{
    int nrs[STUDENT_COUNT] {10, 20, 30, 40, 50, 60, 70, 80}; // 10, 20, 30, 40, 50, 60, 70, 80 (stack array)
    int *p_idnr {new int[STUDENT_COUNT] {0} }; // 0 0 0 0 0 0 0 0 (heap array with exception)
    double *p_score {new(std::nothrow) double[STUDENT_COUNT] {0} }; // 0 0 0 0 0 0 0 0 (heap array without exception)
    double *p_salary {new(std::nothrow) double[STUDENT_COUNT] {1, 2, 3, 4} }; // 1 2 3 4 0 0 0 0 (heap array without exception)

    // stack array supports sizeof(nrs)
    std::cout << "sizeof(nrs) " << sizeof(nrs) << std::endl;

    // stack array supports std::size(nrs) (nrof elements)
    std::cout << "std::size(nrs) " << std::size(nrs) << std::endl;

    // stack array supports range based loop
    for (int value : nrs)
    {
        std::cout << value << std::endl;
    }

    // heap array supports sizeof(p_idnr)
    std::cout << "sizeof(p_idnr) " << sizeof(p_idnr) << std::endl;

    // heap array does not support std::size(p_idnr) (nrof elements)
    // std::cout << "std::size(p_idnr) " << std::size(p_idnr) << std::endl;

    // heap array does not support range based loop
    // for (int value : p_idnr)
    // {
    //     std::cout << value << std::endl;
    // }

    for (size_t i {0}; i < STUDENT_COUNT; i++)
    {
        std::cout << i << " *p_idnr elements " << p_idnr[i] << " " << *(p_idnr + i) << std::endl;
        std::cout << i << " *p_score elements " << p_score[i] << " " <<  *(p_score + i) << std::endl;
        std::cout << i << " *p_salary elements " << p_salary[i] << " " << *(p_salary + i) << std::endl;
    }

    delete[] p_idnr; // free heap array
    p_idnr = nullptr;

    delete[] p_score; // free heap array
    p_score = nullptr;

    delete[] p_salary; // free heap array
    p_salary = nullptr;

    return (0);
}
